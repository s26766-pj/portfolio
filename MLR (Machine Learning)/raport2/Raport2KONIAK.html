def plot_clusters(X, labels, title='Cluster plot'):
    unique_labels = set(labels)
    colors = plt.cm.tab10(np.linspace(0, 1, len(unique_labels)))
    
    plt.figure(figsize=(8, 6))
    for lbl, col in zip(sorted(unique_labels), colors):
        mask = (labels == lbl)
        if lbl == -1:
            plt.scatter(X[mask, 0], X[mask, 1],
                        c='k', marker='x', s=50, label='Szum (-1)')
        else:
            plt.scatter(X[mask, 0], X[mask, 1],
                        c=[col], marker='o', s=50, label=f'Klaster {lbl}')
    
    plt.title(title)
    plt.xlabel('PC1')
    plt.ylabel('PC2')
    plt.legend(loc='best')
    plt.grid(True)
    plt.tight_layout()
    plt.show()

Opis algorytmu DBSCAN (Density-Based Spatial Clustering of Applications with Noise):

  • jest to algorytm klasteryzacji
  • grupuje punkty na podstawie ich gstoci w przestrzeni
  • punkty dzielone s na rdzeniowe, brzegowe i szum
  • kluczowe s dwa parametry: eps (promie ssiedztwa) i min_samples (minimalna liczba punktw)
  • wykrywa nieliniowe klastry i automatycznie redukuje szumy
  • nie wymaga podania liczby klastrw z gry, wyznacza je na podstawie struktury gstoci danych
  • nie wyznacza centroidw

def compute_purity(true_labels, pred_labels):
    cm = confusion_matrix(true_labels, pred_labels)
    max_in_clusters = cm.max(axis=0)
    purity = max_in_clusters.sum() / np.sum(cm)
    return purity, cm
  • przepuszczenie wynikiw algorytmw klasyfikujcych przez powysz funkcj oraz wydrukowanie wartoci czystoci (purity)
